140 glm::vec3 p1 = m_vertices[m_indices[i + 0]].position;

143
144 float v321 = p3.x * p2.y * p1.z;
145 float v231 = p2.x * p3.y * p1.z;

150 m_volume += (1.0f / 6.0f) * (-v321 + v231 + v312 - v132 - v213 + v123);
